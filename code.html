Let \( S_t \) be the price of the asset at time \(t\). Consider a European call option that matures at time \(T\). We use a discrete time model with \(n\) steps to approximate the evolution of asset price over the time interval \([0,T]\). The binomial model assumes that in one time period (after one time step with size \(\delta_t\)), the asset price will go from the current price \(S\) to either a higher value \(u\cdot S \) ("up") with probability \(p\) or to a lower value \(d\cdot S\) ("down") with probability \(1-p\), where \(p\) is the risk neutral probability. The Cox-Ross-Rubinstein model assumes that \(u = \exp(\sigma\sqrt{\delta_t})\), \(d = \exp(-\sigma\sqrt{\delta_t})\), \(p = \frac{R-d}{u-d}\), where \(\sigma\) is the volatility of the asset and \(R = \exp(r\delta_t)\) is a factor that accounts for compounding riskless interest rate \(r\) over a period of time \(\delta_t\). Let \(K \) be the strike price of the option. We know that at expiration (maturity), the price of an European call option is \( \max(S_T-K,0)\). The one-period (one time step) risk neutral European call option price \(C\), as a function of the asset price \(S\) and time \(t\), is given by the formula \[C(t,S) =\frac{1}{R} \bigg[ p C(t+\delta_t, uS) + (1-p)C(t+\delta_t,dS) \bigg].\] Denote by \(C_j^i\) the call option price \(C(i\delta_t,u^{i-j}d^jS_0)\), at time \(t= i\delta_t\) and asset price \(u^{i-j}d^jS_0\). We first calculate the option price at expiration for the \(n+1\) possible price values (see the asset price tree at \(t = T\)), and iteratively calculate the option price at the previous time step using the one-period pricing formula until we reach the current time 0. That is, the call option price at expiration time \(T = n\delta_t\), given each possible asset price, is: \[C^n_0 = \max(u^nS_0-K,0)\] \[C^n_1 = \max(u^{n-1}dS_0-K,0)\] \[...\] \[C^n_{n-1} = \max(ud^{n-1}S_0-K,0)\] \[C^n_n = \max(d^nS_0-K,0)\] Going back one step, at time \(t = T-\delta_t = (n-1)\delta_t\), the call option price (as a function of the asset price) is: \[C^{n-1}_0 =\frac{1}{R} \bigg[ p C^n_0 + (1-p)C^n_1 \bigg]\] \[C^{n-1}_1 =\frac{1}{R} \bigg[ p C^n_1 + (1-p)C^n_2 \bigg]\] \[...\] \[C^{n-1}_{n-1} =\frac{1}{R} \bigg[ p C^n_{n-1} + (1-p)C^n_n \bigg]\] And so on. Every time we go back one step in time, the number of possible call option prices reduces by one. As we reach time \(t=0\), the value \(C^0_0\) is the price of the European call option given the current (i.e. \(t=0\)) stock price \(S\). The European put option price can be easily calculated using put-call parity: \[P = C -S + K \exp(-r(T-0)).\]

#ifndef option_h
#define option_h
using namespace std;
#include <math.h>       // exp, sqrt, pow
struct option {
    double call;
    double put;
};

option optionPrice
(
 double S, // stock price
 double K, // strike price
 double r, // riskless interest rate
 double t, // current time
 double T, // expiration time
 double sig, // volatility
 int n // number of steps
)
{
    option result;

    // calculate step size
    double dt = (T-t)/n;
    // use CRR model for u, d, p
    double u = exp(sig*sqrt(dt));
    double d = 1/u;
    double R = exp(r*dt);
    double p = (R-d)/(u-d);

    // allocate (n+1) nodes for option values
    double * V = new double [n+1];

    // calculate the option prices at time T
    for (int j = 0; j < n+1; j++){
        V[j] = max(S*pow(u,j)*pow(d,n-j) - K, 0.);
    }

    // iteratively calculate the option price at the previous time step until time is t
    // using one period pricing formula
    for (int i = n-1; i >= 0; i--){
        for (int j = 0; j<= i; j++){
            V[j] = (1/R)*(p*V[j+1] + (1-p)*V[j]);
        }
    }

    // set value for call price
    result.call = V[0];
    // calculate put price from put-call parity
    // P = C - S +  K*exp(-r(T-t))
    result.put = result.call - S + K*exp(-r*(T-t));
    delete [] V;
    return result;
};
#endif

#ifndef american_option_h
#define american_option_h
using namespace std;
#include<vector>
#include <math.h>       // exp, sqrt, pow
struct option {
  double call;
  double put;
};

// European option calculator
// Code in Tutorial 1 above
option optionPrice();

double americanOptionPrice
(
 double S, // stock price
 double K, // strike price
 double r, // riskless interest rate
 double t, // current time
 double T, // expiration time
 double sig, // volatility
 int n, // number of steps
 vector<double> divdates, // date of dividends
 double divp // yield for each dividend
)
{
    double call;

    // calculate step size
    double dt = (T-t)/n;
    // use CRR model for u, d, p
    double u = exp(sig*sqrt(dt));
    double d = 1/u;
    double R = exp(r*dt);
    double p = (R-d)/(u-d);
    double ex_t = 0;

    // if no dividends, call the European call option function
    if (divdates.size() == 0){
        option result = optionPrice( S, K, r, t, T, sig, n);
        return result.call;
    }

    // build asset prices tree
    double ** S_tree = new double * [n+1];

    int i = 0;
    int j = 0;
    // allocate arrays
    for(i=0; i<n+1; i++) {
        S_tree[i] = new double[i+1];
    }

    // find time steps for dividend payment
    for (i=0;i<divdates.size(); i++){
        divdates[i] = divdates[i]/dt;
    }
    int count = 0;
    // calculate price values
    for(i = 0; i< n+1;i++){
        // count how many time dividend is paid
        for(int k = 0; k< divdates.size(); k++){
            if(i>=divdates[k]){
                count = count +1;
            }
        }
        for(j = 0; j<i+1; j++){
            // multiply asset price by the dividend discount factor
            S_tree[i][j] = S*pow(u,j)*pow(d,i-j)*pow(1-divp,count);
        }
        count = 0;
    }

    // allocate (n+1) nodes for option values
    // American call option
    double * V = new double [n+1];
    // European: for comparison
    double * W = new double [n+1];

    // calculate the option prices at time T
    for (j = 0; j < n+1; j++){
        V[j] = max(S_tree[n][j] - K, 0.);
        W[j] = max(S_tree[n][j] - K, 0.);
    }

    // iteratively calculate the option price at the previous time step until time is t
    // using one period pricing formula
    double hj;
    double Cj;
    for (i = n-1; i >= 0; i--){
        for (j = 0; j<= i; j++){
            // payoff for exercising the option
            hj = max(S_tree[i][j] - K, 0.);
            // payoff for keeping the option
            Cj = (1/R)*(p*V[j+1] + (1-p)*V[j]);
            // European price for comparison
            W[j] = (1/R)*(p*W[j+1] + (1-p)*W[j]);
            // American price
            if (hj>= Cj){
                V[j] = hj;
                // record stopping time
                ex_t = i*dt;
            }
            else{
                V[j] = Cj;
            }
        }
    }
    // optimal stopping time is the smallest t such that hj >= Cj
    cout << "Exercise the option at time " << ex_t<< '\n';
    // set value for call price
    call = V[0];
    cout << "European option: "<< W[0]<<'\n';
    // free dynamically allocated memory
    delete [] V;
    delete [] W;
    for(i=0; i<n+1; i++) {
        delete [] S_tree[i];
    }
    delete [] S_tree;
    return call;
};
#endif
